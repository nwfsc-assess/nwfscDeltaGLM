Sim_List = SimSimple(
  MeanEncounter = 0.5,
  Nyears = 10,
  Nstrata = 15,
  Nvessels = 4,
  Obsperyear = 175,
  sigmaV = rep(0, 2),
  sigmaVY = rep(1, 2),

X.pos = matrix( rnorm(nrow(masterDat)*nX.pos, mean=0, sd=1), ncol=nX.pos)
